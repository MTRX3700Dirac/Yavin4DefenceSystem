363 void panTiltISR(void)
364 {
365  unsigned int timer_value;
366  static Delay current_delay;
367  unsigned char i = 0;
368 
369  if (CCP2_INT)
370  {
371  timer_value = ReadTimer3();
372 
373  if (timer_value > PWM_PERIOD)
374  {
375  IN_PWM_PIN = 1;
376  WriteTimer3(0); //Clear timer2
377  current_delay = global_delay; //update the static delay
378  changed = 1; //Indicate the change has been loaded
379  OpenCompare2(COM_INT_ON & COM_UNCHG_MATCH, current_delay.InclinationDelay);
380  }
381  else if (timer_value > current_delay.AzimuthDelay)
382  {
383  AZ_PWM_PIN = 0;
384  OpenCompare2(COM_INT_ON & COM_UNCHG_MATCH, PWM_PERIOD);
385  }
386  else if (timer_value > PWM_HALF_PERIOD)
387  {
388  AZ_PWM_PIN = 1;
389  OpenCompare2(COM_INT_ON & COM_UNCHG_MATCH, current_delay.AzimuthDelay);
390  }
391  else if (timer_value > current_delay.InclinationDelay)
392  {
393  IN_PWM_PIN = 0;
394  OpenCompare2(COM_INT_ON & COM_UNCHG_MATCH, PWM_HALF_PERIOD);
395  }
396 
397  PIR2bits.CCP2IF = 0;
398  }

416 static Direction delay2Direction(Delay dly)
417 {
418  Direction ret;
419 
420  ret.azimuth = ((dly.AzimuthDelay - DUTY_CYCLE_TIME - PWM_HALF_PERIOD) * (long int)arcRange.azimuth + DIV_2(DUTY_CYCLE_TIME)) / DUTY_CYCLE_TIME;
421  ret.inclination = ((dly.InclinationDelay - DUTY_CYCLE_TIME + LATENCY) * (long int)arcRange.inclination + DIV_2(DUTY_CYCLE_TIME))/ DUTY_CYCLE_TIME;
422 
423  ret.azimuth = ret.azimuth - DIV_2(arcRange.azimuth) - calibration_offset.azimuth;
424  ret.inclination = -(ret.inclination - DIV_2(arcRange.inclination) - calibration_offset.inclination);
425 
426  return ret;
427 }
